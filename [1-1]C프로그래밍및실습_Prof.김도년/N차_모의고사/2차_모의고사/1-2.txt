#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;

	int max, min;
	int is_first = 1;

	while (1)
	{
		scanf("%d", &n);

		if (n == 0)	break;

		//대칭한 수 만들기
		int result = n;
		int edit_n = n;
		while (edit_n > 0)
		{
			int digit = edit_n % 10;
			edit_n /= 10;
			result = (result * 10) + digit;
		}

		int edit_result = result;
		int three = 0;
		while (edit_result > 0)
		{
			int digit = edit_result % 10;
			edit_result /= 10;

			if (digit == 3)	three++;
		}
		
		//max, min 초기값 설정 / 입력받는 수 범위를 몰라서
		if (is_first)
		{
			is_first = 0;
			max = result;
			min = result;
		}
		
		if (max < result)
		{
			max = result;
		}
		if (min > result)
		{
			min = result;
		}

		printf("%d %d ", result, three);
	}
	printf("\n%d %d", min, max);

	return 0;
}
