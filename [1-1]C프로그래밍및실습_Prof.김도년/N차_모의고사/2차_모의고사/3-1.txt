#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n, m;
	scanf("%d %d", &n, &m);

	int prime = 0;

	for (int i = 0; i < n; i++)
	{
		
		int num;
		scanf("%d", &num);

		//소수인지 확인
		int found = 0;
		for (int j = 2; j < num; j++)
		{
			if (num % j == 0)	found = 1;
		}

		// 입력받은 수가 1일 때 제외
		if (num == 1)	found = 1;

		//소수일 때
		if (!found)
		{
			prime++;
			if (prime <= m)
			{
				printf(" %d", num);
			}
			//소수가 m개를 초과했을 때
			else
			{
				break;
			}
		}
	}

	//소수가 없을 때
	if (!prime)
	{
		printf("0");
	}

	return 0;
}
