#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	scanf("%d", &n);

	int cnt = 0, max_cnt = 0;
	int prime;
	int last_prime;

	int is_last = 0;

	for (int i = 0; i < n; i++)
	{
		if (i == n - 1)	is_last = 1;
		int num;
		scanf("%d", &num);

		int found = 0;
		for (int j = 2; j < num; j++)
		{
			if (num % j == 0)	found = 1;
		}
		if (num == 1)	found = 1;
		if (num >= 100)	found = 1;

		if (!found)
		{
			cnt++;
			prime = num;
		}
		else
		{
			if (max_cnt < cnt)
			{
				max_cnt = cnt;
				last_prime = prime;
			}
			cnt = 0;
		}
	}
	if (is_last)
	{
		if (max_cnt < cnt)
		{
			max_cnt = cnt;
			last_prime = prime;
		}
	}

	if (max_cnt == 0)
	{
		printf("0");
	}
	else
	{
		printf("%d\n%d", max_cnt, last_prime);
	}

	return 0;
}
