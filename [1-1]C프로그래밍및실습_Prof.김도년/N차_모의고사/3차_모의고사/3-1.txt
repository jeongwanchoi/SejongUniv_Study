#include <stdio.h>

int main() {
    int N, x[100];
    int i, j;
    double avg = 0;
    
    scanf("%d", &N);
    
    for (i = 0; i < N; i++)  scanf("%d", &x[i]);
    
    for (i = 0; i < N; i += 3) {
        if (i + 2 < N) {
            avg = (double)(x[i] + x[i + 1] + x[i + 2]) / 3;
            for (j = 0; j < 3; j++) {
                if (x[i + j] >= avg)  printf(" %d", x[i + j]); 
            }
        }
        else if (i + 2 == N) {
            avg = (double)(x[i] + x[i + 1]) / 2;
            for (j = 0; j < 2; j++) {
                if (x[i + j] >= avg)  printf(" %d", x[i + j]);
            }
        }
        else if (i + 1 == N) {
            avg = x[i];
            if (x[i] >= avg)  printf(" %d", x[i]);
        }
    }
    return 0;
}
