#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int reverse_num(int n);

int main()
{
	int n;
	scanf("%d", &n);

	int result = reverse_num(n);
	printf("%d", result);

	return 0;
}

int reverse_num(int n)
{
	int result = 0;
	while (n > 0)
	{
		int digit = n % 10;
		n /= 10;
		result = (result * 10) + digit;
	}

	return result;
}
