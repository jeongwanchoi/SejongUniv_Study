#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int prime_number(int num);
int odd_number(int num);
int count_number(int cnt, int num);

int main()
{
	int n;
	int x[100];
	int x_idx = -1;
	int max_cnt = 0, cnt = 0;
	int last_idx = -1;
	
	while (1)
	{
		scanf("%d", &n);
		if (n <= 0)	break;

		if (prime_number(n) || odd_number(n))
		{
			x[++x_idx] = n;
		}
	}

	for (int i = 0; i <= x_idx; i++)
	{
		cnt = count_number(cnt, x[i]);
		if (max_cnt < cnt)
		{
			max_cnt = cnt;
			last_idx = i;
		}
	}
	if (last_idx == -1)	printf("0");
	else
	{
		for (int i = last_idx - max_cnt + 1; i <= last_idx; i++)
		{
			printf("%d ", x[i]);
		}
	}

	return 0;
}

int prime_number(int num)
{
	int n = num;
	int found = 0;
	for (int i = 2; i < n; i++)
	{
		if (n % i == 0)	found = 1;
	}
	if (n == 1)	found = 1;

	if (!found)	return 1;
	else	return 0;
}
int odd_number(int num)
{
	int n = num;
	int found = 0;
	while (n > 0)
	{
		int digit = n % 10;
		n /= 10;
		if (digit % 2 == 0)	found = 1;
	}
	if (!found)	return 1;
	else	return 0;
}
int count_number(int cnt, int num)
{
	int n = num;

	if (prime_number(n) && odd_number(n))	return cnt + 1;
	else	return 0;
}
