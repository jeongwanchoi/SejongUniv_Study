#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int a, b, c, d, e;
	scanf("%d %d %d %d %d", &a, &b, &c, &d, &e);

	if (a < 0 || a>100 ||
		b < 0 || b>100 ||
		c < 0 || c>100 ||
		d < 0 || d>100 ||
		e < 0 || e>100)
	{
		printf("Invalid");
	}
	else
	{
		//평균 구하기
		double avg = (a + b + c + d + e) / 5.0;

		//최솟값 찾기
		int min = 101;
		if (min > a)	min = a;
		if (min > b)	min = b;
		if (min > c)	min = c;
		if (min > d)	min = d;
		if (min > e)	min = e;

		//40점미만 성적 카운트
		int cnt = 0;
		if (a < 40)	cnt++;
		if (b < 40)	cnt++;
		if (c < 40)	cnt++;
		if (d < 40)	cnt++;
		if (e < 40)	cnt++;

		if (avg >= 90)
		{
			if (cnt >= 3)
			{
				printf("F\navg=%.2lf min=%d", avg, min);
			}
			else
			{
				printf("A+\navg=%.2lf min=%d", avg, min);
			}
		}
		else if (avg >= 75)
		{
			if (cnt >= 3)
			{
				printf("F\navg=%.2lf min=%d", avg, min);
			}
			else
			{
				printf("A\navg=%.2lf min=%d", avg, min);
			}
		}
		else if (avg >= 65)
		{
			if (cnt >= 3)
			{
				printf("F\navg=%.2lf min=%d", avg, min);
			}
			else
			{
				printf("B+\navg=%.2lf min=%d", avg, min);
			}
		}
		else if (avg >= 50)
		{
			if (cnt >= 3)
			{
				printf("F\navg=%.2lf min=%d", avg, min);
			}
			else
			{
				printf("B\navg=%.2lf min=%d", avg, min);
			}
		}
		else
		{
			printf("F\navg=%.2lf min=%d", avg, min);
		}
	}


	return 0;
}
