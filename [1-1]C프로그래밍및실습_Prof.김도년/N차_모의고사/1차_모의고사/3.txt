#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	char input_ch;
	char first = 'z' + 1, second = 'z' + 1;

	while (1)
	{
		scanf("%c", &input_ch);
		if (input_ch >= 'a' && input_ch <= 'z')
		{
			//input_ch < first < second
			if (input_ch < first)
			{
				second = first;
				first = input_ch;
			}
			//first < input_ch < second
			else if(input_ch > first && input_ch < second)
			{
				second = input_ch;
			}
		}
		else
		{
			break;
		}
	}
	printf("%c%c", first, second);

	return 0;
}
