#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	scanf("%d", &n);


	// % 3d
	for (int i = 1; i <= 31; i++)
	{
		if (i == 1)
		{
			for (int j = 1; j <= n; j++)
			{
				printf("   ");
			}
			printf("% 3d", i);
		}
		else
		{
			printf("% 3d", i);
		}
		
		//줄바꿈 설정
		if ((n + i) % 7 == 0)
		{
			printf("\n");
		}
	}

	return 0;
}
