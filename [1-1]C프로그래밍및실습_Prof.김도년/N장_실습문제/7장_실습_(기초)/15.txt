#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	char ch[10];
	for (int i = 0; i < 10; i++)	scanf("%c", &ch[i]);

	int cnt[10] = { 0 };
	for (int i = 0; i < 10; i++)
	{
		for (int j = 0; j < 10; j++)
		{
			if (i != j && ch[i] == ch[j])
			{
				cnt[i]++;
			}
		}
	}

	for (int i = 0; i < 10; i++)
	{
		if (cnt[i] == 0)
		{
			printf(" %c", ch[i]);
		}
	}

	return 0;
}
