#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int a, b;

	while (1)
	{
		scanf("%d %d", &a, &b);
		if (a == 0 || b == 0)	break;


		if (a >= b)
		{
			for (int i = a; i > 0; i--)
			{
				if (a % i == 0 && b % i == 0)
				{
					printf("%d\n", i);
					break;
				}
			}
		}
		else if (a < b)
		{
			for (int i = b; i > 0; i--)
			{
				if (a % i == 0 && b % i == 0)
				{
					printf("%d\n", i);
					break;
				}
			}
		}
	}

	return 0;
}
