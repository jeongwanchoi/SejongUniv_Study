#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() {
    int n1, n2, i, j, num = 1;
    scanf("%d %d", &n1, &n2);
    for (i = n1; i <= n2; i++) {
        printf("%d!=1", i);
        num = 1;
        for (j = 2; j <= i; j++) {
            num *= j;
            printf("*%d", j);
        }
        printf("=%d\n", num);
    }
}
