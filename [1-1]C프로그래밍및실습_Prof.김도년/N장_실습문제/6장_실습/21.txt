#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main() {
   int n;
   int sum = 0;
   int cnt = 0;
   double avg = 0;

   while (1) {
      scanf("%d", &n);
      if (n <= 0) {
         break;
      }
      else if (n > 100) {
         continue;
      }
      else {
         cnt += 1;
         sum += n;
      }
   }
   avg = (double)sum / cnt;
   printf("%d %.2f %d", sum, avg, cnt);
   return 0;
}
