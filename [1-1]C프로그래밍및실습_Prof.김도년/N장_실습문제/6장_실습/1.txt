#define _CRT_SECURE_NO_WARNINGS // for ignoring warning alert

#include <stdio.h>

int main()
{
	int n, i = 0; // two integer variables
	scanf("%d", &n); //input

	while (i <= n) // divisor check
	{
		i++;

		if (n % i == 0)
		{
			printf("%d ", i); //print
		}
	}

	return 0; //return 0
}
