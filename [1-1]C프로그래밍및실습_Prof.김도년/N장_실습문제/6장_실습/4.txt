#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
// while문으로작성했습니다.

int main() {
   int n, m, i=1, x=1;
   scanf("%d %d", &n, &m);

   while (i <= n && i <= m) {
      if (n % i == 0 && m % i == 0) {
         x = i;
      }
      i += 1;
   }

   printf("%d", x);
   return 0;

}
