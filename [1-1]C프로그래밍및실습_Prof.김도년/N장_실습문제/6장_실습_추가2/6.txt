#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int median;
	int a, b, c;
	
	while (1)
	{
		scanf("%d %d %d", &a, &b, &c);
		if (a <= 0 || b <= 0 || c <= 0)	break;

		median = (((b >= a && a >= c) || (c >= a && a >= b)) * a) + (((a >= b && b >= c) || (c >= b && b >= a)) * b) + (((a >= c && c >= b) || (b >= c && c >= a)) * c);

		printf("%d\n", median);
	}

	return 0;
}
