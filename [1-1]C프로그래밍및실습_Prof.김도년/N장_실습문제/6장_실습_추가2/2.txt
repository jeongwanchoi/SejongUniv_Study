#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	int change = 0;

	int p_m; //plus:1, minus:0
	int first_input = 1;
	while (1)
	{
		scanf("%d", &n);

		if (n == 0)	break;

		if (first_input)
		{
			p_m = n > 0 ? 1 : 0;
			first_input = 0;
		}
		else
		{
			if (p_m == 1 && n < 0)
			{
				printf("\n");
				change++;
				p_m = 0;
			}
			else if (p_m == 0 && n > 0)
			{
				printf("\n");
				change++;
				p_m = 1;
			}
		}
		printf("%d ", n);
	}
	printf("\ncount = %d", change);

	return 0;
}
