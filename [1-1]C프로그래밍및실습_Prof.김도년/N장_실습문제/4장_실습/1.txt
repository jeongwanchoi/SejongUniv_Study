#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	int ttho, tho, hun, ten, one;

	scanf("%d", &n);

	ttho = n / 10000;
	tho = (n % 10000) / 1000;
	hun = (n % 1000) / 100;
	ten = (n % 100) / 10;
	one = n % 10;

	printf("%dTTHO %dTHO %dHUN %dTEN %d", ttho, tho, hun, ten, one);

	return 0;
}
