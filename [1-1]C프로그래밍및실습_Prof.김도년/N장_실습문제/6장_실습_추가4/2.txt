#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	char ch;
	while (1)
	{
		scanf("%c", &ch);
		if (ch == '*')
		{
			printf("*");
			break;
		}
		
		if (ch >= 'a' && ch <= 'z')
		{
			printf("%c", ch - 'a' + 'A');
		}
		else
		{
			printf("%c", ch);
		}

	}

	return 0;
}
