#include <stdio.h>

int main() {
    int a, b, c;
    int M, N;
    scanf("%d %d", &M, &N);
    
    for (int i = 1; i < 7; i++) {
        for (int j = 1; j < 7; j++) {
            for (int k = 1; k < 7; k++) {
                int s = i + j + k;
                if (M <= s && s <= N)  printf(" %d %d %d\n", i, j, k);
            }
        }
    }
    
    return 0;
}
