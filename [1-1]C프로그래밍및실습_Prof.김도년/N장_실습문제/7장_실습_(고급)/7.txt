#define _CRT_SECURE_NO_WARNINGS // ignoring warning messages

#include <stdio.h>

int main()
{
	int nums[15], non_dup[15], cnt[15] = {0}, len = 0;

	for (int i = 0; i < 10; i++)	scanf("%d", &nums[i]);

	for (int i = 0; i < 10; i++)
	{
		//found-check
		int found = 0;
		for (int j = 0; j < i; j++) 
		{
			if (nums[i] == nums[j])
			{
				found++;
			}
		}
		if (found)	continue; //known number

		else //new one
		{
			//count the number of num
			for (int j = i; j < 10; j++)
			{
				if (nums[i] == nums[j])
				{
					cnt[len]++;
				}
			}
			non_dup[len] = nums[i];
			len++;
		}
	}

	for (int i = 0; i < len; i++)
	{
		printf("%d %d\n", non_dup[i], cnt[i]);
	}

	return 0;
}
