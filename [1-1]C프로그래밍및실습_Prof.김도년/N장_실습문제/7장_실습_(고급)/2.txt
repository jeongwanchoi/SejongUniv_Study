#define _CRT_SECURE_NO_WARNINGS // ignoring warning messages

#include <stdio.h>

int main()
{
	char arr[12]; //dec 1 char-arr
	int t = 0; //dec 1 int-var

	for (int i = 0; i < 7; i++)	scanf("%c", &arr[i]); //input

	for (int i = 0; i < 7; i++)	if (arr[i] == 'c' && arr[i+1] == 'a' && arr[i+2] == 't')	t++; //count "cat"

	printf("%d", t); //print

	return 0;
}
