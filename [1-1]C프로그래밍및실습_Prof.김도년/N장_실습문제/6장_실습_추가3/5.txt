#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	scanf("%d", &n);

	int edit_n = n;
	int rvs = 0;
	while (edit_n > 0)
	{
		int digit = edit_n % 10;
		edit_n /= 10;

		if (digit == 0)	continue;
		rvs = (rvs * 10) + digit;
	}

	int edit_rvs = rvs;
	int result = 0;

	while (edit_rvs > 0)
	{
		int digit = edit_rvs % 10;
		result = (result * 10) + digit;
		edit_rvs /= 10;
	}
	printf("%d", result * result);

	return 0;
}
