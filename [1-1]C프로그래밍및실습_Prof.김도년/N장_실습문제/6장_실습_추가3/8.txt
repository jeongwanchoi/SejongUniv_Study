#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	int is_even = 0;

	int sum = 0;
	while (1)
	{
		scanf("%d", &n);
		if (n <= 0)	break;


		if (n % 2 == 0)
		{
			if (is_even == 1)
			{
				sum += n;
			}
			is_even = 1;
		}
		else
		{
			if (is_even == 1)
			{
				sum += n;
			}
			is_even = 0;
		}
	}
	printf("%d", sum);

	return 0;
}
