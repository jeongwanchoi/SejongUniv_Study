#include <stdio.h>
#define _CRT_SECURE_NO_WARNINGS

void prt(int x[][5], int N); // 원형

int main(void) { 
    int n, a[20][5]; // 변수

    scanf("%d", &n); // 입력
    
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < 5; j++) { // 입력
            scanf("%d", &a[i][j]);
        }
    }

    prt(a, n); // 함수에 넣음
}

void prt(int x[][5], int N) {
    for (int i = N - 1; i >= 0; i--) { // 반대로 출력함
        for (int j = 0; j < 5; j++) { // 이중 for문
            printf("%d ", x[i][j]);
        }

        printf("\n");
    }
}
