#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int arr[3];
	for (int* p = arr; p < arr + 3; p++)
	{
		scanf("%d", p);
	}
	for (int* p = arr; p < arr + 2; p++)
	{
		for (int* q = p + 1; q < arr + 3; q++)
		{
			if (*p > *q)
			{
				int tmp;
				tmp = *p;
				*p = *q;
				*q = tmp;
			}
		}
	}
	printf("%d", *(arr+1));
}
