#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

void swap(int* a, int* b);

int main()
{
	int n;
	scanf("%d", &n);
	
	int num[50], *p_num = num;
	for (p_num = num; p_num < num + n; p_num++)	scanf("%d", p_num);

	int a, b;
	scanf("%d %d", &a, &b);

	swap(&num[a], &num[b]);

	for (p_num = num; p_num < num + n; p_num++)
	{
		printf(" %d", *p_num);
	}

	return 0;
}

void swap(int* a, int* b)
{
	int tmp = *a;
	*a = *b;
	*b = tmp;
}
