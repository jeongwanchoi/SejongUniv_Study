#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	scanf("%d", &n);

	double arr[100];
	for (int i = 0; i < n; i++)
	{
		scanf("%lf", &arr[i]);
	}

	for (int i = 0; i < n - 1; i++)
	{
		for (int j = i + 1; j <= n - 1; j++)
		{
			if (arr[i] < arr[j])
			{
				double tmp = arr[i];
				arr[i] = arr[j];
				arr[j] = tmp;
			}
		}
	}

	double input;
	scanf("%lf", &input);
	int rank = 1;

	for (int i = 0; i < n; i++)
	{
		printf(" %.1lf", arr[i]);
		if (input < arr[i])	rank++;
	}

	printf("\nHis rank is %d", rank);
}
