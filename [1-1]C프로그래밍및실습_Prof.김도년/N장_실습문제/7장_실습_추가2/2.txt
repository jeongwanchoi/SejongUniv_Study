#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	scanf("%d", &n);

	int x[100];
	for (int i = 0; i < n; i++)scanf("%d", &x[i]);

	int l, m;
	scanf("%d %d", &l, &m);

	for (int i = l; i <= m; i++)
	{
		int found = 0;
		for (int j = 2; j < x[i]; j++)
		{
			if (x[i] % j == 0)	found = 1;
		}
		if (x[i] == 1)	found = 1;

		if (!found)
		{
			printf(" %d", x[i]);
		}
	}
}
