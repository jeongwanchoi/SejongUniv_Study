#include<stdio.h>

int main(){

	/* Type or paste your code in this area */
    int N ; // 변수 n 선언
   int even = 0 , odd = 0; // 변수 선언 및 초기화
   scanf("%d" ,&N); // 정수 n 입력 받기
   while(N > 0){
      int a = N % 10; // a에 n의 일의 자리 수 저장
      if (a % 2 == 0) // a가 짝수이면
         even = even * 10 + a; // even에 10을 곱하고 a를 더하여 대입
      else{ // a가 홀수이면
         odd = odd * 10 + a;// odd에 10을곱해서 a를 더하여 대입
      }
      N = N / 10; // n을 10으로 나눈 몫 저장
   }
   printf("%d %d" ,even ,odd); // 결과출력

	return 0;
}
