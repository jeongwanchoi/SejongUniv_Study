#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int s1, s2, s3;
	scanf("%d %d %d", &s1, &s2, &s3);

	int max = -1, min = 101;
	double avg = (s1 + s2 + s3) / 3.0;

	if (avg >= 90)	printf("A\n");
	else if (avg >= 80)	printf("B\n");
	else if (avg >= 70)	printf("C\n");
	else if (avg >= 60)	printf("D\n");
	else	printf("F\n");

	if (max < s1)	max = s1;
	if (max < s2)	max = s2;
	if (max < s3)	max = s3;

	if (min > s1)	min = s1;
	if (min > s2)	min = s2;
	if (min > s3)	min = s3;

	printf("max: %d\nmin: %d", max, min);

	return 0;
}
