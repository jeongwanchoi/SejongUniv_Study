#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	scanf("%d", &n);

	//

	int num[100];
	for (int i = 0; i < n; i++)	scanf("%d", &num[i]);

	//

	for (int i = n-1; i >= 0; i--)	printf(" %d", num[i]);
	printf("\n");
	
	//

	int max_arr[100];
	int min_arr[100];
	int p = n / 3;
	int r = n % 3;

	for (int i = 0; i < p; i++)
	{
		int max = -9999, min = 9999;
		for (int j = 0; j < 3; j++)
		{
			if (max < num[3 * i + j])
			{
				max = num[3 * i + j];
			}
			if (min > num[3 * i + j])
			{
				min = num[3 * i + j];
			}
		}
		max_arr[i] = max;
		min_arr[i] = min;
	}

	for (int i = 0; i < p; i++)
	{
		printf(" %d", max_arr[i]);
	}
	if (n % 3 == 1)
	{
		printf(" %d", num[n - 1]);
	}
	else if (n % 3 == 2)
	{
		if (num[n - 2] >= num[n - 1])	printf(" %d", num[n - 2]);
		else	printf(" %d", num[n - 1]);
	}

	printf("\n");

	for (int i = 0; i < p; i++)
	{
		printf(" %d", min_arr[i]);
	}
	if (n % 3 == 1)
	{
		printf(" %d", num[n - 1]);
	}
	else if (n % 3 == 2)
	{
		if (num[n - 2] <= num[n - 1])	printf(" %d", num[n - 2]);
		else	printf(" %d", num[n - 1]);
	}



	return 0;
}
