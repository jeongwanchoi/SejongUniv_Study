#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int base_arr[100], non_dup[100];
	int num;
	int base_idx = -1, non_dup_idx = -1;

	//입력받기
	while(1)
	{
		scanf("%d", &num);
		
		if (num >= 0)
		{
			base_arr[++base_idx] = num;
		}
		else
		{
			break;
		}
	}

	//하나씩 확인
	for (int i = 0; i <= base_idx; i++)
	{
		//추가하기 전 인덱스 하나하나 중복 확인 출력
		int base_found = 0;
		for (int j = 0; j <= non_dup_idx; j++)
		{
			if (base_arr[i] == non_dup[j])	base_found = 1;
		}

		if (!base_found) //new one
		{
			printf("1");
		}
		else //dup
		{
			printf("0");
		}
		
		//중복체크
		int found = 0;
		for (int j = 0; j <= non_dup_idx; j++)
		{
			if (base_arr[i] == non_dup[j])
			{
				found = 1;
				break;
			}
		}

		//추가
		if (!found) //새로운 것
		{
			non_dup[++non_dup_idx] = base_arr[i];
		}
	}

	return 0;
}
