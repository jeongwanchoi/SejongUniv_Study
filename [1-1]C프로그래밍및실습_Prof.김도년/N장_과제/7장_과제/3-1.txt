#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int A[25], m; // dec 1 int-arr, 1 int-var

	for (int i = 0; i < 20; i++)	A[i] = i+1; //setting
	scanf("%d", &m); 

	int tmp = A[19]; //store
	for (int i = 19; i > 20-m; i--) //shift
	{
		A[i] = A[i - 1];
	}

	A[20 - m] = tmp; //store

	for (int i = 0; i < 20; i++)	printf("%d ", A[i]);


	return 0;
}
