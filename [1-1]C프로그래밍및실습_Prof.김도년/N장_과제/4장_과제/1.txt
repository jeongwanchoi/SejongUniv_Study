#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int n;
	scanf("%d", &n);

	int Q = (n % 15 == 0) * 15 + (n % 3 == 0 && n % 5 != 0) * 3 + (n % 5 == 0 && n % 3 != 0) * 5 + (n % 3 != 0 && n % 5 != 0) * n;
	// cf) (T:1 or F:0) * smth

	printf("%d is a multiple of %d.", n, Q);

	return 0;
}
