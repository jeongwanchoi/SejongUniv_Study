#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int a, b, c;
	scanf("%d %d %d", &a, &b, &c);

	int x = a / 100, y = b / 100, z = c / 100;

	char Q = (x == y && y == z) * 'T' + ((x == y && y != z) || (y == z && z != x) || (z == x && x != y)) * 'D' + (x != y && y != z && z != x)*'S';

	printf("%c", Q);

	return 0;
}
