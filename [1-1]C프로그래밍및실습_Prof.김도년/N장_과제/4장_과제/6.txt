#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>

int main()
{
	int a, b, c;
	scanf("%d %d %d", &a, &b, &c);

	int Q =
		((a >= b && a <= c) || (a >= c && a <= b)) * a + ((b >= a && b <= c) || (b >= c && b <= a)) * b + ((c >= a && c <= b) || (c >= b && c <= a)) * c;

	printf("%d", Q);

	return 0;
}
